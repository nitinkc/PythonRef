# Dictionary uses {}

#if 'z' in dict:

str = str.split() # By Default it splits with whitespace
d = {} #Declaring Dictionary
for s in str:
    if (s in d):
        d[s] += 1
    else:
        d[s] = 1
        
print (d)

# Tuples play a sort of "struct" role in Python
